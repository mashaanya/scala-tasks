import scala.math.pow
import scala.math.sqrt
class Rational (val n:Double, val d:Double) {
  require(d != 0)
  private def gcd(a:Double,b:Double):Double = {
    if (b==0) a else gcd(b,a%b)
  }
  private val NOD = gcd(n.abs,d.abs)
  val numer = n/NOD
  val denom = d/NOD

  override def toString = numer + "/" + denom

  def +(that: Rational) = new Rational(
    numer * that.denom + that.numer * denom, denom * that.denom
  )
  def -(that: Rational) = new Rational(
    numer * that.denom - that.numer * denom, denom * that.denom
  )
  def *(that: Rational) = new Rational(
    numer * that.numer, denom * that.denom
  )
  def /(that: Rational) = new Rational(
    numer * that.denom, denom * that.numer
  )
  def this(n:Int) = this(n,1)
  def *(that:Int)= new Rational(numer* that,denom)
  def +(that:Int)= new Rational(numer + that*denom, denom)
  def /(that:Int)= new Rational(numer, denom * that)
  def -(that:Int)= new Rational(numer-that*denom, denom)
  def **(n:Int)= new Rational(pow(numer,n), pow(denom,n))
  def ss(that:Rational)= new Rational(sqrt(that.numer),sqrt(that.denom))

}

print(a.ss(a))
